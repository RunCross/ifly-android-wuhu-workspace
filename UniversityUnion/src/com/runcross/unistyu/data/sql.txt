--（1）用户注册及登录模块；（2）整合腾讯微博模块；（3）失物招领模块；
--（4）新闻及公告通知模块；（5）用户活动发布及召集模块；（6）校园地理信息查询模块； (7)用户信息模块

--登录表
create table user_login(
  uid int primary key auto_increment,
  login_name varchar(50) ,
  user_pwd varchar(12)
);

--组织
create table union(
  union_top_id int primary key auto_increment,
  union_superior int default (0),--上级组织，为空则没有,若有则是top_id的值
  union_name varchar(100),
  union_address varchar(250),
  union_contact varchar(250),
  union_introduction text  
);

--用户基本信息表
create table user_info(
  uid int primary key,
  user_name varchar(50),
  gender char(1),
  user_email varchar(55),
  user_union int, --目前所在学校
  user_union_history varchar(255), --曾经所在学校
  user_address varchar(255),
  user_contact varchar(255)
  
  
  CONSTRAINT `user_union_uid` FOREIGN KEY (`user_union`) REFERENCES `union` (`union_id`)
   CONSTRAINT `user_login_uid` FOREIGN KEY (`uid`) REFERENCES `user_login` (`uid`)
);
--新闻
--DATETIME 8 1000-01-01 00:00:00/9999-12-31 23:59:59 YYYY-MM-DD HH:MM:SS 混合日期和时间值 
create table news(
  news_id int primary key auto_increment,
  news_author int ,
  news_union int,
  news_title varchar(50),
  news_date DATETIME default sysdate(), 
  news_visits int , --访问次数
  news_discuss char(1) default '0',  --1 运行评论，0 不允许评论
  news_content text
);
--评论
create table discuss(
  diss_id int primary key auto_increment,
  news_id int,
  diss_author int,
  diss_date DATETIME default sysdate(),
  diss_content text
);
